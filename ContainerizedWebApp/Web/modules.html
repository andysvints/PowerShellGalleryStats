    Get module(s) information and scores.

    Get-PSGalleryStats -Module PipeScript
#>

(
    [Parameter(Mandatory=$true, 
            ValueFromPipeline=$true,
            ValueFromPipelineByPropertyName=$true, 
            Position=0)]
    $Module
)
